    <script>

        // 初始化烟花效果

        const canvas = document.getElementById('fireworksCanvas');

        const fireworks = new Fireworks.default(canvas, {

            speed: 4,

            opacity: 0.8,

            width: window.innerWidth,

            height: window.innerHeight

        });

        fireworks.start();


        // 监听用户输入的文字

        const input = document.getElementById('customText');

        input.addEventListener('keypress', function(event) {

            if (event.key === 'Enter') {

                document.querySelector('h1').textContent = input.value || '欢迎来到烟花特效页面!';

                input.value = '';

            }

        });

    </script>
